#### basic recursion
```
def fib(index):
  if index == 0:
    return 0
  
  if index == 1:
    return 1
  
  if index < 0:
    raise Exception("invalid index", index)
  
  return fib(index-1)+fib(index-2)

  
  
print(fib(0))
print(fib(3))
#print(fib(-1))
print(fib(10))
```
